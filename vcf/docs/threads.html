[

Explain how to use threads by either deriving from VCF::Thread with a new thread class, or by implementing Runnable and passing in the Runnable instance. i.e. 

      class MyThread : public VCF::Thread {
      public
         virtual bool run(); //...implement however you want
      };
      
      //in some code to start the thread
      VCF::Thread * thread = new MyThread();
      thread->start();  //executes the MyThread::run()

or 

      class SomeObject : public ..., public VCF::Runnable {
      public:
      		//methods....
      		//implement Runnable's run() method
      		virtual bool run() ;
      };
	//in some code to start the thread
	SomeObject* someObjInstance = ... //get the instance from somewhere
	VCF::Thread* thread = new VCF::Thread( someObjInstance );
	thread->start(); //executes the SomeObject::run()
